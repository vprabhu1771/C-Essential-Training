// standard input and output header file
#include<stdio.h>

// console input and output header file
#include<conio.h>

// main function
void main()
{
	int x = 10, y = 10, result;

	// clear previous output
	clrscr();


	printf("x = %d", x);

	printf("y = %d", y);
	
	printf("%d >= %d\n", x, y);


	// direct result printing	
	printf("%d\n", x>=y);


	// store result then printing
	result = x >= y;

	printf("%d\n", result);
	

	// get output window front to user
	getch();
}
